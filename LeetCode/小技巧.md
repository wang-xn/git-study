public class Pair implements Comparable{
        int a;
        int b;
        public Pair(int a,int b){
            this.a=a;
            this.b=b;
        }
        @Override
        public int compareTo(Object o1){
            Pair o=(Pair)o1;
            if(this.a!=o.a){
                return this.a-o.a;
            }else{
                return this.b-o.b;
            }
        }
    }
